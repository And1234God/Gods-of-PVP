} const gameBoard = document.getElementById("game-board"); const startButton = document.getElementById("start-game"); const waveNumberDisplay = document.getElementById("wave-number"); const enemiesLeftDisplay = document.getElementById("enemies-left"); // Start with 5 zombies } function spawnZombies(count) { enemies = []; enemiesLeftDisplay.textContent = count; for (let i = 0; i < count; i++) { const zombie = document.createElement("div"); zombie.classList.add("zombie"); zombie.style.top = `${Math.random() * 250}px`; // Random position zombie.style.left = `${Math.random() * 450}px`; zombie.addEventListener("click", () => killZombie(zombie)); gameBoard.appendChild(zombie); enemies.push(zombie); } } function killZombie(zombie) { gameBoard.removeChild(zombie); enemies = enemies.filter((z) => z !== zombie); enemiesLeftDisplay.textContent = enemies.length; if (enemies.length === 0) { nextWave(); } } function nextWave() { if (waveNumber < maxWaves) { waveNumber++; waveNumberDisplay.textContent = waveNumber; spawnZombies(5 + waveNumber * 2); // Increase zombie count each wave } else { alert("You survived all waves!"); } }
